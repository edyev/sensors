cmake_minimum_required(VERSION 3.5)
project(basecamp_service LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

add_executable(basecamp_service
    basecamp_service/src/main.cpp
    basecamp_service/src/req_rep.cpp
)

source_group("Source Files" FILES basecamp_service/src/main.cpp)

# set warning levels for various platforms
if ( CMAKE_CXX_COMPILER_ID MATCHES "Clang|AppleClang|GNU" )
    target_compile_options(basecamp_service PRIVATE -Wall -Wextra -Wunreachable-code -Wpedantic)
endif()
if ( CMAKE_CXX_COMPILER_ID MATCHES "Clang" )
    target_compile_options(basecamp_service PRIVATE -Wweak-vtables -Wexit-time-destructors -Wglobal-constructors -Wmissing-noreturn)
endif()
if ( CMAKE_CXX_COMPILER_ID MATCHES "MSVC" )
    target_compile_options(basecamp_service PRIVATE /W4 /w44265 /w44061 /w44062)
endif()
